24 ** brief: parse the file to get the instructions of champions and informations
25 ** @champion: our structure for each champion
26 ** @file: name of the file given by the user
27 ** return: {0 if succeed} {1 if not a corewar executable or problem with file}
28 */
29 static unsigned int my_champion_file(t_champion *champion, const char *file)
30 {
31  int fd;
32  char integer[4];
33 
34  if ((fd = open(file, O_RDONLY)) == -1)
35  return (my_fprintf(2, "Can't open \"%s\".\n", file));
36  if (read(fd, integer, 4) < 4)
37  return (my_fprintf(2, "Not enough character for magic in \"%s\".\n", file));
38  if (SWAP_ENDIAN(*((int*)integer)) != COREWAR_EXEC_MAGIC)
39  return (my_fprintf(2, "Magic number is wrong in \"%s\".\n", file));
40  if (read(fd, champion->name, PROG_NAME_LENGTH) < PROG_NAME_LENGTH)
41  return (my_fprintf(2, "Not enough characters for name in \"%s\".\n", file));
42  champion->name[PROG_NAME_LENGTH] = '\0';
43  if (read(fd, integer, 4) < 4)
44  return (my_fprintf(2, "Not enough characters for size in \"%s\".\n", file));
45  champion->size = SWAP_ENDIAN(*((int*)integer));
46  if (read(fd, champion->comment, COMMENT_LENGTH) < COMMENT_LENGTH)
47  return (my_fprintf(2, "Not enough characters for comm in \"%s\".\n", file));
48  champion->comment[COMMENT_LENGTH] = '\0';
49  champion->code = my_malloc(champion->size * sizeof(champion->code));
50  if (read(fd, champion->code, champion->size) < champion->size)
51  return (my_fprintf(2, "Not enough characters for prog in \"%s\".\n", file));
52  champion->live = 0;
53  close(fd);
54  return (0);
55 }
